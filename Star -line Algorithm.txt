#include <GL/glut.h>
#include <iostream>
#include <cmath>

int round_val(float v) {
  return floor(v + 0.5);
}

void lineDDA(int x1, int y1, int x2, int y2) {
    float dx = x2 - x1;
    float dy = y2 - y1;
    int steps;

    if (abs(dx) > abs(dy)) {
        steps = abs(dx);
    } else {
        steps = abs(dy);
    }

    float x_inc = dx / (float)steps;
    float y_inc = dy / (float)steps;

    float x = x1;
    float y = y1;

    glBegin(GL_POINTS);
    for (int i = 0; i <= steps; i++) {
        glVertex2i(round_val(x), round_val(y));
        x += x_inc;
        y += y_inc;
    }
    glEnd();
}

void lineBresenham(int x1, int y1, int x2, int y2) {
    int dx = abs(x2 - x1);
    int dy = abs(y2 - y1);
    int sx = (x1 < x2) ? 1 : -1;
    int sy = (y1 < y2) ? 1 : -1;
    int err = dx - dy;

    glBegin(GL_POINTS);
    while (true) {
        glVertex2i(x1, y1);

        if (x1 == x2 && y1 == y2) break;

        int e2 = 2 * err;
        if (e2 > -dy) {
            err -= dy;
            x1 += sx;
        }
        if (e2 < dx) {
            err += dx;
            y1 += sy;
        }
    }
    glEnd();
}

void drawStar() { 
    int p0[] = {250, 450};
    int p1[] = {393, 359};
    int p2[] = {359, 219};
    int p3[] = {141, 219};
    int p4[] = {107, 359};

    lineBresenham(p0[0], p0[1], p2[0], p2[1]);
    lineBresenham(p2[0], p2[1], p4[0], p4[1]);
    lineBresenham(p4[0], p4[1], p1[0], p1[1]);
    lineBresenham(p1[0], p1[1], p3[0], p3[1]);
    lineBresenham(p3[0], p3[1], p0[0], p0[1]);
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);
    glColor3f(1.0, 1.0, 0.0); 
    drawStar();
    glFlush();
}

void init() {
    glClearColor(0.0, 0.0, 0.2, 1.0); 
    gluOrtho2D(0, 500, 0, 500);
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(500, 500);
    glutInitWindowPosition(150, 150);
    glutCreateWindow("Star using Line Algorithms");
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
