#include <GL/glut.h>
#include <iostream>
#include <cmath>

int round_val(float v) {
  return floor(v + 0.5);
}

void lineDDA(int x1, int y1, int x2, int y2) {
    float dx = x2 - x1;
    float dy = y2 - y1;
    int steps;

    if (abs(dx) > abs(dy)) {
        steps = abs(dx);
    } else {
        steps = abs(dy);
    }

    float x_inc = dx / (float)steps;
    float y_inc = dy / (float)steps;

    float x = x1;
    float y = y1;

    glBegin(GL_POINTS);
    for (int i = 0; i <= steps; i++) {
        glVertex2i(round_val(x), round_val(y));
        x += x_inc;
        y += y_inc;
    }
    glEnd();
}

void lineBresenham(int x1, int y1, int x2, int y2) {
    int dx = abs(x2 - x1);
    int dy = abs(y2 - y1);
    int sx = (x1 < x2) ? 1 : -1;
    int sy = (y1 < y2) ? 1 : -1;
    int err = dx - dy;

    glBegin(GL_POINTS);
    while (true) {
        glVertex2i(x1, y1);

        if (x1 == x2 && y1 == y2) break;

        int e2 = 2 * err;
        if (e2 > -dy) {
            err -= dy;
            x1 += sx;
        }
        if (e2 < dx) {
            err += dx;
            y1 += sy;
        }
    }
    glEnd();
}

void drawLine(int x1, int y1, int x2, int y2) {
    lineBresenham(x1, y1, x2, y2);
}

void drawA(int x, int y) {
    drawLine(x, y, x + 20, y + 80);
    drawLine(x + 20, y + 80, x + 40, y);
    drawLine(x + 10, y + 40, x + 30, y + 40);
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);
    glColor3f(1.0, 1.0, 1.0);
    drawA(20, 300);
    drawE(80, 300);
    drawF(140, 300);
    drawH(200, 300);
    drawI(260, 300);
    drawK(320, 300);
    drawL(380, 300);

    drawM(20, 150);
    drawN(80, 150);
    drawT(140, 150);
    drawV(200, 150);
    drawX(260, 150);
    drawY(320, 150);
    drawZ(380, 150);

    glFlush();
}

void init() {
    glClearColor(0.0, 0.0, 0.0, 1.0);
    gluOrtho2D(0, 500, 0, 500);
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(500, 500);
    glutInitWindowPosition(100, 100);
    glutCreateWindow("Alphabets using Line Algorithms");
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
