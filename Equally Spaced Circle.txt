#include <GL/glut.h>
#include <cmath>

int xc = 250, yc = 250;

void plotPoints(int x, int y, int xc, int yc)
{
    glBegin(GL_POINTS);
        glVertex2i(xc + x, yc + y);
        glVertex2i(xc - x, yc + y);
        glVertex2i(xc + x, yc - y);
        glVertex2i(xc - x, yc - y);
        glVertex2i(xc + y, yc + x);
        glVertex2i(xc - y, yc + x);
        glVertex2i(xc + y, yc - x);
        glVertex2i(xc - y, yc - x);
    glEnd();
}

void drawCircle(int xc, int yc, int r)
{
    int x = 0, y = r;
    int p = 1 - r;

    while (x <= y)
    {
        plotPoints(x, y, xc, yc);

        if (p < 0)
            p = p + 2 * x + 1;
        else
        {
            p = p + 2 * (x - y) + 1;
            y--;
        }

        x++;
    }
}

void display()
{
    glClear(GL_COLOR_BUFFER_BIT);

    int r = 40;
    int dist = 120;

    glColor3f(1, 0, 0);          // Center red circle
    drawCircle(xc, yc, r);

    for (int i = 0; i < 6; i++)
    {
        float angle = i * 60 * 3.14159 / 180.0f;
        int x = xc + dist * cos(angle);
        int y = yc + dist * sin(angle);

        switch (i)
        {
            case 0: glColor3f(0, 1, 0); break;
            case 1: glColor3f(0, 0, 1); break;
            case 2: glColor3f(1, 1, 0); break;
            case 3: glColor3f(1, 0, 1); break;
            case 4: glColor3f(0, 1, 1); break;
            case 5: glColor3f(1, 0.5, 0); break;
        }

        drawCircle(x, y, r);
    }

    glFlush();
}

void init()
{
    glClearColor(0, 0, 0, 1);
    glPointSize(2);

    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();

    gluOrtho2D(0, 500, 0, 500);
}

int main(int argc, char** argv)
{
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(500, 500);
    glutCreateWindow("Equally Spaced Circles");

    init();
    glutDisplayFunc(display);
    glutMainLoop();

    return 0;
}
