#include <GL/glut.h>
#include <stdio.h>

void plot_point(int x1, int y1, int x, int y)
{
    glBegin(GL_POINTS);
        glVertex2i(x1 + x, y1 + y);
        glVertex2i(x1 - x, y1 + y);
        glVertex2i(x1 + x, y1 - y);
        glVertex2i(x1 - x, y1 - y);
        glVertex2i(x1 + y, y1 + x);
        glVertex2i(x1 - y, y1 + x);
        glVertex2i(x1 + y, y1 - x);
        glVertex2i(x1 - y, y1 - x);
    glEnd();
}

void bresenham_circle(int x1, int y1, int r)
{
    int x = 0;
    int y = r;
    int p = 3 - 2 * r;

    plot_point(x1, y1, x, y);

    while (x < y)
    {
        x++;
        if (p < 0)
        {
            p = p + 4 * x + 6;
        }
        else
        {
            y--;
            p = p + 4 * (x - y) + 10;
        }
        plot_point(x1, y1, x, y);
    }

    glFlush();
}

void Draw_circle()
{
    glClear(GL_COLOR_BUFFER_BIT);
    int x1 = 300, y1 = 250, r = 125;
    bresenham_circle(x1, y1, r);
}

int main(int argc, char** argv)
{
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowPosition(0, 0);
    glutInitWindowSize(640, 480);
    glutCreateWindow("Bresenham Circle Algorithm");

    glClearColor(0.0, 0.0, 0.0, 0.0);
    glColor3f(0.0, 1.0, 1.0);
    gluOrtho2D(0, 640, 0, 480);

    glutDisplayFunc(Draw_circle);
    glutMainLoop();

    return 0;
}
