#include <GL/glut.h>
#include <cmath>
#include <iostream>

using namespace std;

// ------------------------------------------------------
// Draw a single pixel
// ------------------------------------------------------
void setPixel(int x, int y)
{
    glBegin(GL_POINTS);
        glVertex2i(x, y);
    glEnd();
}

// ------------------------------------------------------
// Plot 8 symmetric points of the circle
// ------------------------------------------------------
void plotCirclePoints(int xc, int yc, int x, int y)
{
    setPixel(xc + x, yc + y);
    setPixel(xc - x, yc + y);
    setPixel(xc + x, yc - y);
    setPixel(xc - x, yc - y);

    setPixel(xc + y, yc + x);
    setPixel(xc - y, yc + x);
    setPixel(xc + y, yc - x);
    setPixel(xc - y, yc - x);
}

// ------------------------------------------------------
// Midpoint Circle Algorithm
// ------------------------------------------------------
void midpointCircle(int xc, int yc, int r)
{
    int x = 0;
    int y = r;
    int p = 1 - r;

    while (x <= y)
    {
        plotCirclePoints(xc, yc, x, y);

        if (p < 0)
            p = p + 2 * x + 3;
        else
        {
            p = p + 2 * (x - y) + 5;
            y--;
        }

        x++;
    }
}

// ------------------------------------------------------
// Display
// ------------------------------------------------------
void display()
{
    glClear(GL_COLOR_BUFFER_BIT);

    glColor3f(1, 0, 0);   // Red circle
    midpointCircle(250, 250, 120);

    glFlush();
}

// ------------------------------------------------------
// Initialize 2D viewing area
// ------------------------------------------------------
void init()
{
    glClearColor(0, 0, 0, 1);    
    gluOrtho2D(0, 500, 0, 500);  // Set coordinate range
}

// ------------------------------------------------------
// Main
// ------------------------------------------------------
int main(int argc, char** argv)
{
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(500, 500);
    glutInitWindowPosition(100, 100);
    glutCreateWindow("Midpoint Circle Algorithm (C++)");

    init();
    glutDisplayFunc(display);
    glutMainLoop();

    return 0;
}
