#include <iostream>
#include <GL/glut.h>

using namespace std;

void myInit();
void plotPixel(int x, int y, float* fill_color);
void boundaryFill(int x, int y, float* fill_color, float* boundary_color);
void polygonShape();
int main(int argc, char **argv);

float boundary_color[3];
float fill_color[3];

void myInit() {
    glOrtho(0.0, 500.0, 0.0, 500.0, -1.0, 1.0);
    glClearColor(0.0, 0.0, 0.0, 1.0);
    glMatrixMode(GL_PROJECTION);
    glColor3f(0.0, 0.0, 1.0);
}

void plotPixel(int x, int y, float* color) {
    glColor3fv(color);
    glBegin(GL_POINTS);
    glVertex2i(x, y);
    glEnd();
    glFlush();
}

void boundaryFill(int x, int y, float* fill_color, float* boundary_color) {
    float current[3];

    glReadPixels(x, y, 1, 1, GL_RGB, GL_FLOAT, current);

    if (
        (current[0] != boundary_color[0] || current[1] != boundary_color[1] || current[2] != boundary_color[2]) &&
        (current[0] != fill_color[0] || current[1] != fill_color[1] || current[2] != fill_color[2])
    ) {
        plotPixel(x, y, fill_color);

        boundaryFill(x + 1, y, fill_color, boundary_color);
        boundaryFill(x - 1, y, fill_color, boundary_color);
        boundaryFill(x, y + 1, fill_color, boundary_color);
        boundaryFill(x, y - 1, fill_color, boundary_color);
    }
}

void polygonShape() {
    glClear(GL_COLOR_BUFFER_BIT);

    glColor3f(0.0, 0.0, 1.0);

    glBegin(GL_LINE_LOOP);

    glVertex2i(100, 100);
    glVertex2i(250, 100);
    glVertex2i(250, 200);
    glVertex2i(100, 200);

    glEnd();
    glFlush();

    boundary_color[0] = 0.0;
    boundary_color[1] = 0.0;
    boundary_color[2] = 1.0;

    fill_color[0] = 1.0;
    fill_color[1] = 1.0;
    fill_color[2] = 0.0;

    boundaryFill(102, 102, fill_color, boundary_color);
}

int main(int argc, char **argv) {
    glutInit(&argc, argv);
    glutInitWindowSize(500, 500);
    glutInitWindowPosition(800, 150);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutCreateWindow("Boundary Fill");
    myInit();
    glutDisplayFunc(polygonShape);
    glutMainLoop();

    return 0;
}
