#include <GL/glut.h>
#include <cmath>

int windowWidth = 800;
int windowHeight = 300;

void drawCircle(int xc, int yc, int radius) {
    glBegin(GL_LINE_LOOP);
    for (int i = 0; i <= 360; i++) {
        float theta = i * 3.14159 / 180;
        int x = xc + radius * cos(theta);
        int y = yc + radius * sin(theta);
        glVertex2i(x, y);
    }
    glEnd();
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);
    glColor3f(0, 0, 0);

    int radius = 60;
    int y_center = 150;
    int x_start = 150;
    int gap = 110;

    drawCircle(x_start, y_center, radius);
    drawCircle(x_start + gap, y_center, radius);
    drawCircle(x_start + 2 * gap, y_center, radius);
    drawCircle(x_start + 3 * gap, y_center, radius);

    glFlush();
}

void init() {
    glClearColor(1, 1, 1, 1);
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluOrtho2D(0, windowWidth, 0, windowHeight);
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(windowWidth, windowHeight);
    glutCreateWindow("Audi Logo - 4 Rings");
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
