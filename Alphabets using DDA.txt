#include <GL/glut.h>
#include <iostream>
#include <cmath>

using namespace std;

void putPixel(int x, int y) {
    glBegin(GL_POINTS);
        glVertex2i(x, y);
    glEnd();
}

void DDA_Line(int x1, int y1, int x2, int y2) {
    int dx = x2 - x1;
    int dy = y2 - y1;
    int steps = abs(dx) > abs(dy) ? abs(dx) : abs(dy);

    float x_inc = dx / (float)steps;
    float y_inc = dy / (float)steps;

    float x = x1;
    float y = y1;

    for (int i = 0; i <= steps; i++) {
        putPixel(round(x), round(y));
        x += x_inc;
        y += y_inc;
    }
}

void drawA(int x, int y) {
    DDA_Line(x, y, x + 20, y + 40);
    DDA_Line(x + 40, y, x + 20, y + 40);
    DDA_Line(x + 10, y + 20, x + 30, y + 20);
}

void drawE(int x, int y) {
    DDA_Line(x, y, x, y + 40);
    DDA_Line(x, y + 40, x + 30, y + 40);
    DDA_Line(x, y + 20, x + 20, y + 20);
    DDA_Line(x, y, x + 30, y);
}

void drawF(int x, int y) {
    DDA_Line(x, y, x, y + 40);
    DDA_Line(x, y + 40, x + 30, y + 40);
    DDA_Line(x, y + 20, x + 20, y + 20);
}

void drawH(int x, int y) {
    DDA_Line(x, y, x, y + 40);
    DDA_Line(x + 30, y, x + 30, y + 40);
    DDA_Line(x, y + 20, x + 30, y + 20);
}

void drawI(int x, int y) {
    DDA_Line(x + 15, y, x + 15, y + 40);
}

void drawK(int x, int y) {
    DDA_Line(x, y, x, y + 40);
    DDA_Line(x + 30, y, x, y + 20);
    DDA_Line(x, y + 20, x + 30, y + 40);
}

void drawL(int x, int y) {
    DDA_Line(x, y, x, y + 40);
    DDA_Line(x, y, x + 30, y);
}

void drawM(int x, int y) {
    DDA_Line(x, y, x, y + 40);
    DDA_Line(x + 30, y, x + 30, y + 40);
    DDA_Line(x, y + 40, x + 15, y);
    DDA_Line(x + 30, y + 40, x + 15, y);
}

void drawN(int x, int y) {
    DDA_Line(x, y, x, y + 40);
    DDA_Line(x + 30, y, x + 30, y + 40);
    DDA_Line(x, y + 40, x + 30, y);
}

void drawT(int x, int y) {
    DDA_Line(x + 15, y, x + 15, y + 40);
    DDA_Line(x, y + 40, x + 30, y + 40);
}

void drawV(int x, int y) {
    DDA_Line(x, y + 40, x + 15, y);
    DDA_Line(x + 15, y, x + 30, y + 40);
}

void drawW(int x, int y) {
    DDA_Line(x, y + 40, x + 10, y);
    DDA_Line(x + 10, y, x + 20, y + 20);
    DDA_Line(x + 20, y + 20, x + 30, y);
    DDA_Line(x + 30, y, x + 40, y + 40);
}

void drawX(int x, int y) {
    DDA_Line(x, y, x + 30, y + 40);
    DDA_Line(x, y + 40, x + 30, y);
}

void drawY(int x, int y) {
    DDA_Line(x, y + 40, x + 15, y + 20);
    DDA_Line(x + 30, y + 40, x + 15, y + 20);
    DDA_Line(x + 15, y, x + 15, y + 20);
}

void drawZ(int x, int y) {
    DDA_Line(x, y + 40, x + 30, y + 40);
    DDA_Line(x + 30, y + 40, x, y);
    DDA_Line(x, y, x + 30, y);
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);

    drawA(10, 300);
    drawE(70, 300);
    drawF(130, 300);
    drawH(190, 300);
    drawI(250, 300);
    drawK(310, 300);
    drawL(370, 300);
    drawM(430, 300);
    drawN(490, 300);

    drawT(10, 200);
    drawV(70, 200);
    drawW(130, 200);
    drawX(190, 200);
    drawY(250, 200);
    drawZ(310, 200);

    glFlush();
}

void init() {
    glClearColor(1.0, 1.0, 1.0, 1.0);
    glColor3f(0.0, 0.0, 0.0);
    glPointSize(2.0);
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluOrtho2D(0.0, 500.0, 0.0, 500.0);
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(500, 500);
    glutInitWindowPosition(100, 100);
    glutCreateWindow("Characters using DDA Line Algorithm");
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
