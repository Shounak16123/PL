#include <GL/glut.h>
#include <cmath>

void drawLineBresenham(int x1, int y1, int x2, int y2) {
    int dx = abs(x2 - x1);
    int dy = abs(y2 - y1);
    int x = x1;
    int y = y1;

    int xInc = (x2 > x1) ? 1 : -1;
    int yInc = (y2 > y1) ? 1 : -1;

    glBegin(GL_POINTS);
    glVertex2i(x, y);

    if (dx >= dy) {
        int p = 2 * dy - dx;
        for (int i = 0; i < dx; i++) {
            x += xInc;
            if (p < 0) {
                p += 2 * dy;
            } else {
                y += yInc;
                p += 2 * (dy - dx);
            }
            glVertex2i(x, y);
        }
    } else {
        int p = 2 * dx - dy;
        for (int i = 0; i < dy; i++) {
            y += yInc;
            if (p < 0) {
                p += 2 * dx;
            } else {
                x += xInc;
                p += 2 * (dx - dy);
            }
            glVertex2i(x, y);
        }
    }
    glEnd();
}

void drawHouse()
{
    drawLineBresenham(100, 100, 300, 100);
    drawLineBresenham(100, 100, 100, 250);
    drawLineBresenham(300, 100, 300, 250);
    drawLineBresenham(100, 250, 300, 250);

    drawLineBresenham(100, 250, 200, 350);
    drawLineBresenham(300, 250, 200, 350);

    drawLineBresenham(180, 100, 180, 180);
    drawLineBresenham(220, 100, 220, 180);
    drawLineBresenham(180, 180, 220, 180);

    drawLineBresenham(120, 170, 160, 170);
    drawLineBresenham(120, 130, 160, 130);
    drawLineBresenham(120, 130, 120, 170);
    drawLineBresenham(160, 130, 160, 170);

    drawLineBresenham(240, 170, 280, 170);
    drawLineBresenham(240, 130, 280, 130);
    drawLineBresenham(240, 130, 240, 170);
    drawLineBresenham(280, 130, 280, 170);
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);
    glColor3f(1.0, 1.0, 1.0);
    drawHouse();
    glFlush();
}

int main(int argc, char **argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(500, 500);
    glutCreateWindow("House using Bresenham Line Algorithm");
    gluOrtho2D(0, 500, 0, 500);
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
